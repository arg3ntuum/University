#include <stdio.h>
#include <locale.h>
/*Складіть програму, що друкує 1, якщо точка з
координатами (х, у) лежить у середині кола (радіус r, центр у
точці з координатами (а, b)), і 0 — в іншому випадку. Значення
змінних a, b, r, x, y вводять з клавіатури.*/
int main()
{
	setlocale(LC_ALL, "Rus");
	int x = 0, y = 0, a = 0, b = 0, r = 0; //инициализировали, чтобы не было ошибок и это так правильно
	printf("Введите координаты центра круга x y->\n");
	scanf_s("%d %d", &a, &b);
	printf("Введите радиус круга ->\n");
	scanf_s("%d", &r);
	printf("Введите координаты, чтобы проверить их местонахождение(1 - в круге, 0 - вне круга ->\n");
	scanf_s("%d %d", &x, &y);
	if ((x - a) * (x - a) + (y - b) * (y - b) < r * r)
		return 1;
	else
		return 0;
}

#include <stdio.h>

/*Складіть програму, яка порівнює два числа і
друкує повідомлення, яке з чисел більше, або «Числа рівні».
*/
int main()
{
	setlocale(LC_ALL, "Rus");
	int a = 0, b = 0;
	printf("Введите числа для сравнения\n");
	scanf_s("%d %d", &a, &b);
	if (a == b)
		printf("Числа равны друг другу.");
	else if (a > b)
		printf("%d > %d", a, b);
	else
		printf("%d < %d", a, b);
	return 0;
}

#include <stdio.h>
#include <locale.h>
#include <time.h>

/*дана послідовність зі 100 цілих чисел. Визначити кількість чисел у найдовшій послідовності з нулів, розташованих підряд.*/
int main()
{
	setlocale(LC_ALL, "Rus");
	srand(time(NULL));

	int chislo = 0, schetchik = 0, max_poslidovnist = 0;
	
	for (int i = 0;i < 100;i++)
	{
		chislo = 0 + rand() % 2;
		printf("%d ", chislo);
		if (chislo == 0)
			schetchik++;
		else schetchik = 0;
		if (max_poslidovnist < schetchik)
			max_poslidovnist = schetchik;

	}
	printf("\nМаксильмальная последовательность нулей = %d", max_poslidovnist);
	return 0;
}
